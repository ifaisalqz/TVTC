declare
    First_Name varchar(20);
    Major varchar(20);
    Subject varchar(20);

begin
    First_Name:=:First_Name;
    Major:=:Major;
    Subject:=:Subject;
    dbms_output.put_line('Name , ' || First_Name);
    dbms_output.put_line('Major , ' || Major);
    dbms_output.put_line('Subject, ' || Subject);
    dbms_output.put_line(SYSDATE);
END;
